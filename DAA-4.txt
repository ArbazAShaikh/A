def fractional_knapsack():
    # Initialize weights, values, and capacity
    weights=[10,20,30,40]
    values=[20,30,40,50]
    capacity=70
    result = 0  # To store the maximum value we can carry

    # Sort items by value-to-weight ratio in descending order
    items = sorted(zip(weights, values), key=lambda x: x[1] / x[0], reverse=True)

    # Iterate over each item in sorted order
    for weight, value in items:
        if capacity == 0:  # If the knapsack is full, stop
            break

        # If we can take the full weight of the item
        if weight <= capacity:
            result += value  # Add full value
            capacity -= weight  # Reduce capacity by the weight taken
        else:
            # If only part of the item's weight can be taken
            result += (value / weight) * capacity  # Take fraction of the value
            capacity = 0  # Knapsack is now full

    print("Maximum value in knapsack:", result)

if __name__ == "__main__":
    fractional_knapsack()
