def solve_knapsack():
    wt = [3, 4, 6, 5]
    val = [2, 3, 1, 4]
    W = 8
    n = len(val)
    
    # Create a memoization table (2D list) to store results
    memo = [[-1 for _ in range(W + 1)] for _ in range(n + 1)]

    def knapsack(W, n):
        # Base case: if no items left or no capacity left
        if n == 0 or W == 0:
            return 0
        # Check if the result is already computed
        if memo[n][W] != -1:
            return memo[n][W]
        
        # Recursive case: include or exclude the current item
        if wt[n-1] > W:
            result = knapsack(W, n-1)
        else:
            result = max(
                val[n-1] + knapsack(W - wt[n-1], n-1), 
                knapsack(W, n-1)
            )
        
        # Store the result in the memoization table
        memo[n][W] = result
        return result

    print(knapsack(W, n))

solve_knapsack()
